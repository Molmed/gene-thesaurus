# Gene Thesaurus
Translates from older gene names to the current HGNC standard gene symbols.

# Supported lookup fields:
- 'ensembl_gene_id': string
- 'refseq_accession': list
- 'symbol': string
- 'ccds_id': list
- 'omim_id': list
- 'location': string. Format: '8q24.23'
- 'rgd_id': list
- 'merops': string
- 'entrez_id': string
- 'uniprot_ids': list
- 'mgd_id': list
- 'hgnc_id': string
- 'ucsc_id': string
- 'prev_symbol': list
- 'rna_central_id': list
- 'alias_symbol': list
